\begin{mydefinition}
若函数 $f$ 在点 $x_0$ 处满足
\[
\lim_{x\to x_0}f(x)=f(x_0),
\]
则称 $f$ 在 $x_0$ 连续。
\end{mydefinition}

\begin{myexample}

\end{myexample}

\frametitle{命题与证明}

\begin{proposition}
若 $n$ 为偶数，则 $n^2$ 也是偶数。
\end{proposition}

\begin{myproof}
设 $n=2k$，则
\[
n^2=(2k)^2=4k^2=2(2k^2),
\]
故为偶数。
\end{myproof}

\end{frame}
